--#Router Configuration--
--Renaming the Router to (R1)--
enable
configure terminal
hostname R1

--Asings and activates the interface G0/0/0--
interface GigabitEthernet0/0/0
ip address 192.168.1.3 255.255.255.0
no shutdown

--Asings and activates the interface G0/0/1--
interface GigabitEthernet0/0/1
ip address 192.168.2.3 255.255.255.0
no shutdown
